NFE.ID AS ID_VENDA, 
'NFE' AS TIPO, 
NFE.NFE_NUMERO AS NROCUPOM, 
USR.USUARIO, 
NFE.ID_VENDEDOR,
VDR.DESCRICAO AS VENDEDOR, 
NFEI.ITEM, 
NFEI.ID_PRODUTO, 
PROD.PRODUTO, 
NFE.ID_CLIENTE, 
CLI.RAZ_SOCIAL AS CLIENTE, 
NFEI.TOTAL_ITEM AS VLRTOTAL, 
NFEI.QUANTIDADE, 
NFE.DATA_VENDA AS DATACUPOM
FROM PRODUTOS PROD
 
INNER JOIN NFE_ITENS NFEI ON NFEI.ID_PRODUTO = PROD.ID_PRODUTO
INNER JOIN NFE NFE ON NFE.ID = NFEI.ID_NFE
INNER JOIN CLIENTES CLI ON CLI.ID_CLIENTE = NFE.ID_CLIENTE 
INNER JOIN VENDEDOR VDR ON VDR.ID = NFE.ID_VENDEDOR 
INNER JOIN USUARIO USR ON USR.ID = NFE.ID_USUARIO
